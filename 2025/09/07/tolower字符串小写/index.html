tolower字符串小写

string s=”hello”;

//1.转换到新的字符串
string s2;
transform(s.begin(),s.end(),back_inserter(s2),::tolower);//因为新字符串s2的长度未知，所以使用back_inserter(s2)

//2.原地转换
transform(s.begin(),s.end(),s.begin(),::tolower);
